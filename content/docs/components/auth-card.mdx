import { AuthCard } from "@/components/sol/auth-card";

## Installation

<Tabs items={["pnpm", "npm", "yarn"]}>
  <Tab value="pnpm">pnpm dlx solanaui@latest add auth-card</Tab>
  <Tab value="npm">npx solanaui@latest add auth-card</Tab>
  <Tab value="yarn">yarn solanaui@latest add auth-card</Tab>
</Tabs>

## Usage

```tsx
<AuthCard
  title="Sign in"
  showEmail={true}
  socialProviders={[{ name: "Google", icon: <GoogleIcon /> }]}
/>
```
