## Usage

```tsx
const SOL_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/So11111111111111111111111111111111111111112/logo.png";
const USDC_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v/logo.png";

  tokens={[
    { symbol: "SOL", icon: SOL_ICON, price: "$162.56" },
    { symbol: "USDC", icon: USDC_ICON, price: "$1.00" },
  ]}
/>
```

### With Actions

Pass an `actions` array of React nodes to render action buttons for each row. The array should match the length of the `tokens` array.

```tsx
import { Button } from "@/components/ui/button";

const SOL_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/So11111111111111111111111111111111111111112/logo.png";
const USDC_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v/logo.png";

<TokenTable
  tokens={[
    { symbol: "SOL", icon: SOL_ICON, price: "$162.56", apy: "6.82%" },
    { symbol: "USDC", icon: USDC_ICON, price: "$1.00", apy: "8.45%" },
  ]}
  actions={[
    <Button variant="outline" size="sm">Deposit</Button>,
    <Button variant="outline" size="sm">Deposit</Button>,
  ]}
/>
```
