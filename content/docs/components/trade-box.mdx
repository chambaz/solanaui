## Usage

### With Leverage (Perps)

```tsx
const USDC_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v/logo.png";

<TradeBox
  tokens={[{ icon: USDC_ICON, symbol: "USDC" }]}
  defaultToken="USDC"
  balance="4,812.50"
  showLeverage={true}
  leverageMin={1}
  leverageMax={50}
  leverageDefault={5}
  details={[
    { label: "Est. Entry Price", value: "$162.56" },
    { label: "Liquidation Price", value: "$132.45", className: "text-red-400" },
    { label: "Fees", value: "0.05%" },
  ]}
  submitLabel="Open Long"
/>
```

### Without Leverage (Lending)

```tsx
const SOL_ICON =
  "https://raw.githubusercontent.com/solana-labs/token-list/main/assets/mainnet/So11111111111111111111111111111111111111112/logo.png";

<TradeBox
  tokens={[{ icon: SOL_ICON, symbol: "SOL" }]}
  defaultToken="SOL"
  balance="24.58"
  labels={["Deposit", "Withdraw"]}
  defaultSide="long"
  showLeverage={false}
  details={[
    { label: "APY", value: "6.82%" },
    { label: "Utilization", value: "78.4%" },
  ]}
  submitLabel="Deposit SOL"
/>
```
